Expand description

A helper type to construct row vectors. CMatrix is used internally, but makes working with 1-dimensional data less verbose.

const ARRAY: [f64; 2] = [4.0, 7.0];

const ROW_VECTOR: CVector::<2> = CVector::new_vector(ARRAY);
const RESULT: [f64; 2] = ROW_VECTOR.finish_vector();

assert_eq!(ARRAY, RESULT)
